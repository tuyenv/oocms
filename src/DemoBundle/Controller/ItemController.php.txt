<?php

namespace DemoBundle\Controller;

use CoreBundle\CoreFrontController;
use Sensio\Bundle\FrameworkExtraBundle\Configuration\Route;
use Symfony\Component\HttpFoundation\Request;
use Symfony\Component\HttpFoundation\Response;

class ItemController extends CoreFrontController
{
    /**
     * @Route("/item", name="item_page")
     */
    public function itemAction(Request $request)
    {
        return new Response('<h1>Item Page</h1>');
    }

    /**
     * @Route("/item/{itemId}", name="item_detail_page")
     */
    public function itemDetailAction(Request $request, $itemId)
    {
        return new Response('<h1>Item Detail: '.$itemId.'</h1>');
    }
}
